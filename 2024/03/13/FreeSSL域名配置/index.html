2. 进行nginx配置

acme.sh –install-cert -d cloud-disk.lelepen.com \

–key-file /etc/letsencrypt/live/cloud-disk.lelepen.com/privkey.pem \

–fullchain-file /etc/letsencrypt/live/cloud-disk.lelepen.com/fullchain.pem \

–reloadcmd “service nginx force-reload”
